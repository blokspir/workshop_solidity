const { ethers } = require("hardhat");

async function main() {
    // Get deployer account
    const [deployer] = await ethers.getSigners();
    console.log("Deploying contracts with account:", deployer.address);
    
    // Check balance
    const balance = await ethers.provider.getBalance(deployer.address);
    console.log("Account balance:", ethers.formatEther(balance), "ETH");
    
    // Deploy the contract
    console.log("\nüì¶ Deploying FactoringContract...");
    const Factory = await ethers.getContractFactory("FactoringContract");
    const contract = await Factory.deploy();
    
    // Wait for deployment
    await contract.waitForDeployment();
    const address = await contract.getAddress();
    
    console.log("\n‚úÖ FactoringContract deployed successfully!");
    console.log("üìç Contract address:", address);
    console.log("üîç View on Etherscan: https://sepolia.etherscan.io/address/" + address);
    
    // Return address for verification
    return address;
}

main()
    .then(() => process.exit(0))

Step 6: Interact with Your Contract

Now let's run through a complete invoice lifecycle on your deployed contract. Create scripts/interact.js:

const { ethers } = require("hardhat");

const CONTRACT_ADDRESS = "YOUR_DEPLOYED_CONTRACT_ADDRESS";

async function main() {
    const [admin, supplier, buyer] = await ethers.getSigners();
    console.log("Admin:", admin.address);
    
    // Connect to deployed contract
    const contract = await ethers.getContractAt("FactoringContract", CONTRACT_ADDRESS);
    console.log("Connected to contract at:", CONTRACT_ADDRESS);
    
    // 1. Authorize users
    console.log("\nüìù Authorizing supplier...");
    let tx = await contract.authorizeSupplier(supplier.address);
    await tx.wait();
    console.log("‚úÖ Supplier authorized:", supplier.address);
    
    console.log("\nüìù Authorizing buyer...");
    tx = await contract.authorizeBuyer(buyer.address);
    await tx.wait();
    console.log("‚úÖ Buyer authorized:", buyer.address);
    
    // 2. Create invoice (as supplier)
    console.log("\nüìÑ Creating invoice...");
    const amount = ethers.parseEther("10"); // 10 ETH
    const dueDate = Math.floor(Date.now() / 1000) + (30 * 24 * 60 * 60); // 30 days
    tx = await contract.connect(supplier).createInvoice(buyer.address, amount, dueDate);
    await tx.wait();
    console.log("‚úÖ Invoice created!");
    
    // 3. Check invoice
    const invoice = await contract.invoices(1);
    console.log("\nüìã Invoice Details:");
    console.log("   ID:", invoice.id.toString());
    console.log("   Amount:", ethers.formatEther(invoice.amount), "ETH");
    console.log("   Status:", invoice.status.toString(), "(0=Created)");
    
    // 4. Approve invoice (as buyer)
    console.log("\n‚úçÔ∏è Buyer approving invoice...");
    tx = await contract.connect(buyer).approveInvoice(1);
    await tx.wait();
    console.log("‚úÖ Invoice approved!");
    
    console.log("\nüéâ Complete invoice lifecycle executed on Sepolia!");
}

main().catch(console.error);
