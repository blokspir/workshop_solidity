Smart Contract: Access Control

Role-based access control using mappings and modifiers ensures only authorized users can perform specific actions.

// Authorization mappings
mapping(address => bool) public authorizedSuppliers;
mapping(address => bool) public authorizedBuyers;
mapping(address => bool) public authorizedFinanciers;

// Access control modifiers
modifier onlyAdmin() {
    require(msg.sender == admin, "Only admin can call this function");
    _;
}

modifier onlyAuthorizedSupplier() {
    require(authorizedSuppliers[msg.sender], "Not an authorized supplier");
    _;
}

modifier onlyBuyer(uint256 _invoiceId) {
    require(invoices[_invoiceId].buyer == msg.sender, "Not the invoice buyer");
    _;
}

modifier inStatus(uint256 _invoiceId, InvoiceStatus _status) {
    require(invoices[_invoiceId].status == _status, "Invoice not in required status");
    _;
}

// Admin authorization functions
function authorizeSupplier(address _supplier) external onlyAdmin {
    require(_supplier != address(0), "Invalid address");
    authorizedSuppliers[_supplier] = true;
    emit SupplierAuthorizationChanged(_supplier, true);
}

The invoice lifecycle: Create ‚Üí Approve ‚Üí Request Financing ‚Üí Fund ‚Üí Settle

// Create invoice (supplier only)
function createInvoice(address _buyer, uint256 _amount, uint256 _dueDate)
    external onlyAuthorizedSupplier returns (uint256)
{
    require(_buyer != address(0), "Invalid buyer address");
    require(_buyer != msg.sender, "Cannot invoice yourself");
    require(_amount > 0, "Amount must be positive");
    require(_dueDate > block.timestamp, "Due date must be in future");
    
    invoiceCount++;
    invoices[invoiceCount] = Invoice({
        id: invoiceCount,
        supplier: msg.sender,
        buyer: _buyer,
        amount: _amount,
        dueDate: _dueDate,
        status: InvoiceStatus.Created,
        createdAt: block.timestamp,
        approvedAt: 0,
        financier: address(0),
        fundedAmount: 0,
        fundedAt: 0
    });
    
    emit InvoiceCreated(invoiceCount, msg.sender, _buyer, _amount, _dueDate);
    return invoiceCount;
}

// Approve invoice (buyer only)
function approveInvoice(uint256 _invoiceId)
    external invoiceExists(_invoiceId) onlyBuyer(_invoiceId)
    inStatus(_invoiceId, InvoiceStatus.Created)
{
    invoices[_invoiceId].status = InvoiceStatus.Approved;
    invoices[_invoiceId].approvedAt = block.timestamp;
    emit InvoiceApproved(_invoiceId, block.timestamp);
}

Smart Contract: Payment Functions (CEI Pattern)

The Checks-Effects-Interactions pattern prevents reentrancy attacks by updating state before making external calls.

// Financier funds invoice (pays supplier early with discount)
function fund(uint256 _invoiceId) external payable
    invoiceExists(_invoiceId) onlyAuthorizedFinancier
    inStatus(_invoiceId, InvoiceStatus.FinancingRequested)
{
    Invoice storage invoice = invoices[_invoiceId];
    uint256 discountedAmount = calculateDiscountedAmount(invoice.amount);
    require(msg.value >= discountedAmount, "Insufficient funds sent");
    
    // EFFECTS: Update state BEFORE external call (CEI pattern)
    invoice.status = InvoiceStatus.Funded;
    invoice.financier = msg.sender;
    invoice.fundedAmount = discountedAmount;
    invoice.fundedAt = block.timestamp;
    
    // INTERACTIONS: External calls after state updates
    (bool success, ) = payable(invoice.supplier).call{value: discountedAmount}("");
    require(success, "Transfer to supplier failed");
    
    // Refund excess payment
    if (msg.value > discountedAmount) {
        (bool refundSuccess, ) = payable(msg.sender).call{value: msg.value - discountedAmount}("");
        require(refundSuccess, "Refund failed");
    }
    
    emit InvoiceFunded(_invoiceId, msg.sender, discountedAmount);
}

// Calculate discounted amount (e.g., 2% discount = 98% of original)
function calculateDiscountedAmount(uint256 _amount) public view returns (uint256) {
    return _amount - (_amount * discountRate / 10000);
}

Enterprise Implementation Considerations

As you bring blockchain solutions to your clients, keep these enterprise-grade considerations in mind.

Regulatory & Compliance

  • KYC/AML requirements for financial transactions
  • Data privacy regulations (GDPR, PDPA)
  • Securities law for tokenized assets
  • Cross-border transaction rules

Security & Operations

  • Multi-signature wallet governance
  • Key management & HSM integration
  • Disaster recovery procedures
  • Incident response planning

Architecture Decisions

  • Public vs private/consortium chains
  • On-chain vs off-chain data storage
  • Upgradability patterns (proxy contracts)
  • Oracle selection and fallback strategies

Integration

  • ERP system connectivity
  • Legacy database synchronization
  • API gateway and rate limiting
  • Event-driven architecture (indexers)
